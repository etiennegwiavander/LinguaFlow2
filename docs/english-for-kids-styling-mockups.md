## Implementation Notes

### CSS Classes Structure
```typescript
const getKidsStyles = (isKids: boolean) => ({
  container: isKids ? 'space-y-6' : 'space-y-4',
  card: isKids 
    ? 'rounded-2xl border-4 p-6 shadow-xl hover:scale-105 transition-all' 
    : 'rounded-lg border-2 p-4 shadow-sm',
  numberBadge: isKids
    ? 'w-12 h-12 text-xl font-black rounded-full shadow-lg bg-gradient-to-br from-yellow-400 to-orange-400'
    : 'w-8 h-8 text-sm font-bold rounded-full bg-blue-100 text-blue-600',
  text: isKids ? 'text-lg font-semibold' : 'text-base',
  heading: isKids ? 'text-2xl font-black' : 'text-xl font-semibold'
});
```

### Conditional Rendering Pattern
```typescript
const isKidsTemplate = template.category === 'English for Kids';
const styles = getKidsStyles(isKidsTemplate);

return (
  <div className={styles.container}>
    <div className={styles.card}>
      <span className={styles.numberBadge}>1</span>
      <p className={styles.text}>Content</p>
    </div>
  </div>
);
```
